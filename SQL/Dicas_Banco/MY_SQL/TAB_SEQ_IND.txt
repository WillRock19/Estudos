* Tabela
	SELECT
		TABLE_NAME AS TABELA,
		0 AS NR_GERADO
	FROM
		information_schema.TABLES
	WHERE TABLE_SCHEMA = <<DATABASE>>

* Campos, tipo e tamanho
	SELECT
		TABLE_NAME AS TABELA,
		COLUMN_NAME AS CAMPO,
		DATA_TYPE AS TIPO,
		CASE WHEN DATA_TYPE = 'double' OR DATA_TYPE = 'decimal'
		THEN CONCAT('(',NUMERIC_PRECISION,',',NUMERIC_SCALE,')')
		ELSE
			CONCAT('(',CHARACTER_MAXIMUM_LENGTH,')')
		END AS Tamanho,
		CASE WHEN IS_NULLABLE = 'YES' THEN 'NULL' ELSE 'NOT NULL' END AS NAONULL,
		EXTRA AS AUTONUMERACAO      
	FROM information_schema.COLUMNS
	WHERE TABLE_SCHEMA = <<DATABASE>>	

* Sequencia
	SELECT
	      CONCAT(TABLE_NAME,'_s') AS SEQUENCIA,
	      0 AS NR_GERADO 
	FROM information_schema.COLUMNS
	WHERE TABLE_SCHEMA = <<DATABASE>> AND EXTRA = 'auto_increment'

* Index
	SELECT
	      INDEX_NAME AS INDEXES,
	      TABLE_NAME AS TABELA,
	      COLUMN_NAME CAMPO
	      0 AS NR_GERADO
	FROM information_schema.STATISTICS
	WHERE NOT INDEX_NAME LIKE 'PRIMARY%' AND TABLE_SCHEMA = <<DATABASE>>

* Chave primária
	SELECT
	      CONSTRAINT_NAME AS CHAVE,
	      TABLE_NAME AS TABELA,
	      COLUMN_NAME AS CAMPO,
	      0 AS NR_GERADO
	FROM information_schema.KEY_COLUMN_USAGE 
	WHERE CONSTRAINT_SCHEMA = <<DATABASE>>
